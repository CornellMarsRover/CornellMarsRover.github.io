86/* An array in toml. You can extract value/table/array using an index. */
87class Array
88{
89public:
90 // Content kind

98
99 // Return the #elements in the array

101

105 pair<bool, int64_t> getInt(int idx) const;
106 pair<bool, double> getDouble(int idx) const;

108
109 std::unique_ptr<Table> getTable(int idx) const;
110 std::unique_ptr<Array> getArray(int idx) const;
