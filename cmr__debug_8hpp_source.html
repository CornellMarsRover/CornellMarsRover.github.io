283template <typename U, typename T>
284auto narrow_cast(T&& t)
285{
286 using ValT = std::remove_reference<T>;
287 if constexpr (std::is_signed_v<ValT> && std::is_unsigned_v<U>) {
288 CMR_ASSERT_MSG(t >= 0, "Cannot narrow negative value");
289 } else if constexpr (std::is_unsigned_v<ValT> && std::is_signed_v<U>) {
290 CMR_ASSERT_MSG(static_cast<U>(std::forward<T>(t)) >= 0,
291 "Cannot narrow unsigned value");
292 }
293 const auto u = static_cast<U>(std::forward<T>(t));
294 CMR_ASSERT_MSG(static_cast<ValT>(u) == t, "Cannot narrow value");
295 return u;
296}
