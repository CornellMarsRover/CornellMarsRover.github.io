18std::string string_format(const std::string & format, Args ... args)

20 // Source: https://stackoverflow.com/a/26221725/3212994 (explanation commented here too)
21 // the goal of this approach is to create a char buffer of the final string's size and
22 // use std::snprintf to write to it, before finally converting it to a std:string.
23
24 // first, we need to know what the final size actually is.
25 // this requires a call to std::snprintf, where we don't actually write the result anywhere
26 // but just use the size it returns. this also gives us a chance to do some error handling
27 // which is the advantage of this longer-winded approach over std::sprintf.
28 int size_s =
29 std::snprintf(nullptr, 0, format.c_str(), args ...) +
30 1; // add extra space for terminator '\0'
31 if (size_s <= 0) {throw std::runtime_error("string format failed");}
32 auto size = static_cast<size_t>( size_s );
33
34 // now that we know the size, and we know that the format is valid, we finally create our
35 // char buffer and use std::snprintf to write it
36 std::unique_ptr<char[]> buf(new char[size]);
37 std::snprintf(buf.get(), size, format.c_str(), args ...);
38 return {buf.get(), buf.get() + size - 1}; // We don't want the terminator '\0' inside
39}
